965 template <typename T> inline void
966 SCModel<T>::_interpolation
967 ( double*coefmon, const unsigned maxord, const T&X, puniv f )
968 {
969  double b( Op<T>::mid(X) ), a( Op<T>::u(X)-b ), x[maxord+1], fx[maxord+1];
970  double mulconst( PI/(2.*double(maxord+1)) );
971  for( unsigned i(0); i<=maxord; i++ ){
972  x[i] = std::cos(mulconst*(2.*double(i)+1.));
973  fx[i] = f( a*x[i]+b );
974  }
975 
976  switch( maxord ){
977  case 0:
978  coefmon[0] = fx[0];
979  return;
980  case 1:
981  coefmon[0] = 0.5 * ( fx[0] + fx[1] );
982  coefmon[1] = ( fx[1] - fx[0] ) / ( x[1] - x[0] );
983  return;
984  default:
985  for( unsigned i(0); i<=maxord; i++ ){
986  double mulconst2( std::cos(mulconst*double(i)) ),
987  mulconst3( 4*std::pow(mulconst2,2)-2 ),
988  b0( 0 ), b1( 0 );
989  b0 = fx[maxord];
990  b1 = fx[maxord-1] + mulconst3*b0;
991  for( unsigned j=maxord-2; j>1; j-=2 ){
992  b0 = fx[j] + mulconst3*b1 - b0;
993  b1 = fx[j-1] + mulconst3*b0 - b1;
994  }
995  if( !(maxord%2) )
996  b0 = fx[0] + mulconst3*b1 - b0 - b1;
997  else{
998  b0 = fx[1] + mulconst3*b1 - b0;
999  b0 = fx[0] + mulconst3*b0 - b1 - b0;
1000  }
1001  coefmon[i] = 2./double(maxord+1)*mulconst2*b0;
1002  }
1003  coefmon[0] *=0.5;
1004  return;
1005  }
