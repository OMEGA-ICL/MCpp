101  mat_inv.resize(mat.n);

103  char UPLO('l');

124 inline long dgeqrf( const dgematrix& Amat, dgematrix& Qmat, dgematrix& Rmat )
125 {
126  VERBOSE_REPORT;
127  dgematrix A(Amat);
128  const int M(Amat.m), N(Amat.n), NM = (M<N?M:N);
129  dcovector TAU(NM), WORK(M);
130 
131  // Get optimal size for QR decomposition
132  int LWORK(-1), INFO(0);
133  dgeqrf_(&M, &N, A.array, &N, TAU.array, WORK.array, &LWORK, &INFO);
134  LWORK = (int)*WORK.array;
135  WORK.resize(LWORK);
136 
137  // Perform QR decomposition
138  dgeqrf_(&M, &N, A.array, &N, TAU.array, WORK.array, &LWORK, &INFO);
139  if( INFO ){
140  WARNING_REPORT;
141  std::cerr << "DGEQRF: Serious trouble happend. INFO = " << INFO << "." << std::endl;
142  return INFO;
143  }
144 
145  // Post-process QR decomposition results for R
146  Rmat.resize(M,N); Rmat.zero();
147  for( int ir=0; ir<NM; ir++ )
148  for(int ic=ir; ic<N; ic++ )
149  Rmat(ir,ic) = A(ir,ic);
150 #ifdef MC__DEBUG_DGEQRF
151  std::cout << "Matrix R:" << Rmat << std::endl;
152 #endif
153 
154  // Post-process QR decomposition results for Q
155  Qmat.resize(M,N);
156  Qmat.identity();
157  dcovector V(M); V.zero();
158  dgematrix H(M,M);
159 
160  for( int im=0; im<NM; im++ ){
161  V(im) = 1.; if( im ) V(im-1) = 0.;
162  for( int ir=im+1; ir<M; ir++ ) V(ir) = A(ir,im);
163  H.identity(); H -= TAU(im) * V * t(V);
164  Qmat *= H;
165  }
166 #ifdef MC__DEBUG_DGEQRF
167  std::cout << "Matrix Q:" << Qmat << std::endl;
168  std::cout << "Check Q*R:" << Qmat*Rmat << std::endl;
169  std::cout << "=? A:" << Amat << std::endl;
170 #endif
171  return INFO;
172 }
