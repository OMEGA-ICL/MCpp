880  bool operator()
881  ( const FFVar*Var1, const FFVar*Var2 ) const
882  {
883  assert( Var1 && Var2 );
884  // Order variables/constants w.r.t. their types first
885  if( Var1->_id.first < Var2->_id.first ) return true;
886  if( Var1->_id.first > Var2->_id.first ) return false;
887  // If variables, order w.r.t. their index next
888  switch( Var1->_id.first ){
889  case FFVar::VAR:
890  case FFVar::AUX:
891  if( Var1->_id.second < Var2->_id.second ) return true;
892  if( Var1->_id.second > Var2->_id.second ) return false;
893  break;
894  case FFVar::CINT: case FFVar::CREAL:
895  lt_FFNum ltNum;
896  return ltNum( &Var1->_num, &Var2->_num );
897  break;
898  }
899  return false;
