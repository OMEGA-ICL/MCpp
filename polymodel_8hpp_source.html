337 // Based on a code by John Burkardt
338 // http://people.sc.fsu.edu/~jburkardt/cpp_src/monomial/monomial.html
339 inline void
340 PolyModel::_next_expmon
341 ( unsigned *x, const unsigned m ) const
342 {
343  // Ensure that M>=1
344  assert( m );
345 
346  // Find I, the index of the rightmost nonzero entry of X.
347  unsigned i=0, im1, t;
348  for( unsigned j=m; j>=1; j-- ){
349  if ( x[j-1] ){
350  i = j;
351  break;
352  }
353  }
354 
355  // set T = X(I)
356  // set X(I) to zero,
357  // increase X(I-1) by 1,
358  // increment X(D) by T-1.
359  if( !i ){
360  x[m-1] = 1;
361  return;
362  }
363  else if( i == 1 ){
364  t = x[0] + 1;
365  im1 = m;
366  }
367  else{
368  t = x[i-1];
369  im1 = i - 1;
370  }
371 
372  x[i-1] = 0;
373  x[im1-1] = x[im1-1] + 1;
374  x[m-1] = x[m-1] + t - 1;
375 }

395 inline void
396 PolyModel::_ext_expmon
397 ( const unsigned maxord, const bool full )
398 {
399  _ext_binom( maxord );
400  _ext_posord( maxord );
401  delete[] _expmon;
402  if( full ) _expmon_size = std::pow(maxord+1,_nvar);
403  else _expmon_size = _posord[maxord+1];
404  _expmon = new unsigned[ _expmon_size*_nvar];
405  _set_expmon( maxord );
406  if( !full ) return;
407 
408  unsigned *iexp = new unsigned[_nvar];
409  for( unsigned iord=maxord+1, jmon=_posord[maxord+1];
410  jmon<_expmon_size; iord++ ){
411  _ext_binom( iord );
412  _ext_posord( iord );
413  if( _posord[iord] >= _posord[iord+1] )
414  throw typename PolyModel::Exceptions( PolyModel::Exceptions::MAXSIZE );
415  for( unsigned ivar=0; ivar<_nvar; ivar++ ) iexp[ivar] = _expmon[(_nmon-1)*_nvar+ivar];//0;
416  for( unsigned kmon=_posord[iord]; kmon<_posord[iord+1]; kmon++ ){
417  _next_expmon( iexp, _nvar );//iord );
418  bool lt_maxord = true;
419  for( unsigned ivar=0; ivar<_nvar; ivar++ ){
420  if( iexp[ivar] > maxord ){ lt_maxord = false; break; }
421  _expmon[jmon*_nvar+ivar] = iexp[ivar];
422  }
423  if( lt_maxord ){
424 #ifdef MC__POLYMODEL_DEBUG
425  std::cout << jmon << ":";
426  for( unsigned ivar=0; ivar<_nvar; ivar++ )
427  std::cout << " " << _expmon[jmon*_nvar+ivar];
428  std::cout << std::endl;
429 #endif
430  jmon++;
431  if( jmon >= _expmon_size ) break;
432  }
433  }
434  }
435  delete[] iexp;
436  return;
437 }

439 inline unsigned
440 PolyModel::_loc_expmon
441 ( const unsigned *iexp ) const
442 {
443  unsigned ord = 0;
444  for( unsigned i=0; i<_nvar; i++ ) ord += iexp[i];
445 #ifdef MC__POLYMODEL_CHECK
446  assert( ord<_nord+2 );
447 #endif
448  unsigned pos = _posord[ord];
449 
450  unsigned p = _nvar ;
451  for( unsigned i=0; i<_nvar-1; i++ ){
452  p--;
453  for( unsigned j=0; j<iexp[i]; j++ )
454  pos += _get_binom( p-1+ord-j, ord-j );
455  ord -= iexp[i];
456  }
457 
458  return pos;

469  *p = 1;
470  p++;
471  *p = i+1;
472  p++;
473  k = ( i+1<nord? i+1: nord );
474  for( unsigned j=2; j<=k; j++, p++ ) *p = *(p-1) * (i+2-j)/j;
475  for( unsigned j=k+1; j<=nord; j++, p++ ) *p = 0.;
476  }

507 template <typename C, typename U> inline U*
508 PolyModel::_get_bndord
509 ( const C*coefmon, const U*const*bndbasis,
510  const std::set<unsigned>&ndxmon ) const
511 {
512  U* bndord_ = new U[_nord+1];
513  bndord_[0] = (ndxmon.empty() || ndxmon.find(0)!=ndxmon.end())? coefmon[0]: 0.;
514  //bndord_[0] = coefmon[0];
515  for( unsigned i=1; i<=_nord; i++ ){
516  bndord_[i] = 0.;
517  for( auto jt=ndxmon.lower_bound(_posord[i]); jt!=ndxmon.lower_bound(_posord[i+1]); ++jt ){
518  unsigned k=0;
519  const unsigned *jexp = _expmon+(*jt)*_nvar;
520  for( ; k<_nvar; k++ ) if( jexp[k] ) break;
521  U bndmon_ = bndbasis[k][jexp[k]];
522  for( ++k; k<_nvar; k++ ) if( jexp[k] ) bndmon_ *= bndbasis[k][jexp[k]];
523  bndord_[i] += coefmon[*jt] * bndmon_;
524  }
525  for( unsigned j=_posord[i]; ndxmon.empty() && j<_posord[i+1]; j++ ){
526  unsigned k=0;
527  const unsigned *jexp = _expmon+j*_nvar;
528  for( ; k<_nvar; k++ ) if( jexp[k] ) break;
529  U bndmon_ = bndbasis[k][jexp[k]];
530  for( ++k; k<_nvar; k++ ) if( jexp[k] ) bndmon_ *= bndbasis[k][jexp[k]];
531  bndord_[i] += coefmon[j] * bndmon_;
532  }
533  }
534  return bndord_;
535 }

957  if( reset ){
958  for( unsigned imon=0; imon<nmon(); imon++ ) _coefmon[imon] = 0.;
959  *_bndrem = 0.;
960  _ndxmon.clear();
961  }
962 
963  // Copy monomial coefficients from var into *this
964  if( !var._PM ){
965  _coefmon[0] = var._coefmon[0];
966  *_bndrem = *var._bndrem;
967  _bndord_uptd = false;
968  return *this;
969  }
970  unsigned*iexp = new unsigned[nvar()];
971  // Case: sparse representation
972  for( auto it=var._ndxmon.begin();
973  it!=var._ndxmon.end() && *it<var._posord(nord()+1); ++it ){
974  for( unsigned ivar=0; ivar<nvar(); ivar++ )
975  iexp[ivar] = ( ivar<var.nvar()? var._expmon(*it)[ivar]: 0 );
976  //unsigned imon = _loc_expmon(iexp);
977  auto pmon = _ndxmon.insert( _loc_expmon(iexp) );

982  _ndxmon.empty() && jmon<var.nmon() && jmon<var._posord(nord()+1); jmon++ ){
983  for( unsigned ivar=0; ivar<nvar(); ivar++ )

989  *_bndrem += var.bndord(iord);
990  *_bndrem += *var._bndrem;
991  _unset_bndT();
992  _unset_bndpol();
993  _bndord_uptd = false;
994
