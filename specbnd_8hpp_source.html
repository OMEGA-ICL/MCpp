582 template <class T> inline T
583 Specbnd<T>::_LambdaS
584 ( const fadbad::F<T> &a, unsigned int n )
585 {
586  if( !n ) return 0.;
587  if( n == 1 ) return a.size()? Op<T>::sqr( a[0] ): 0.;
588  double upbnd=0.;
589  for (unsigned int i=0; i<n; i++)
590  if( a.size() ) upbnd += Op<T>::u( Op<T>::sqr( a[i] ) );
591  return Op<T>::zeroone() * upbnd;
592 }

594 template <class T> inline T
595 Specbnd<T>::_LambdaT
596 ( const fadbad::F<T> &a, const fadbad::F<T> &b, unsigned int n )
597 {
598  if( !n ) return 0.;
599  if( n == 1 ) return (a.size() && b.size())? 2.*a[0]*b[0]: 0.;
600  double upbnda=0., upbndb=0.;
601  for (unsigned int i=0; i<n; i++){
602  if( a.size() ) upbnda += Op<T>::u( Op<T>::sqr( a[i] ) );
603  if( b.size() ) upbndb += Op<T>::u( Op<T>::sqr( b[i] ) );
604  }
605  T lamb = 2.*(Op<T>::zeroone()-0.5) * std::sqrt(upbnda*upbndb);
606  for (unsigned int i=0; i<n; i++)
607  if( a.size() && b.size() ) lamb += a[i]*b[i];
608  return lamb;
609 }

623  0.5*(D2X[i].deriv(0).deriv(j)+D2X[j].deriv(0).deriv(i)) );
