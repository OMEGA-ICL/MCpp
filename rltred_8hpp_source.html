245 inline typename RLTRED::t_RLTRed&
246 RLTRED::search
247 ( const std::vector<const FFVar*>& vlhs, const bool add2dag )
248 {
249  // Generate list of operation for constraint LHS and particpating variables
250  l_Ops Ops = _dag->subgraph( vlhs );
251 
252  // Define subsets and map of linear operations and participating variables
253  _extract_linearity( Ops, vlhs );
254 
255  // Define subsets of bilinear, fractional, square and square root terms
256  _target_nonlinearity( Ops );
257 
258  // Bi-partite graph approach to identifying reduction constraints
259  switch( options.LEVEL ){
260  case Options::PRIMSEQ: case Options::FULLSEQ: _search_sequential( Ops ); break;
261  case Options::PRIMSIM: case Options::FULLSIM: _search_simultaneous( Ops ); break;
262  }
263 
264  // Add reduced constraints to DAG
265  if( add2dag ) _add_to_dag( vlhs );
266 
267  return _RLTRed;
268 }

270 inline void
271 RLTRED::_add_to_dag
272 ( const std::vector<const FFVar*>& vlhs )
273 {
274  // RLT did not give any reduction constraints
275  _ctrRed.clear();
276  if( _RLTRed.begin() == _RLTRed.end() ){
277  if( options.DISPLAY >= 1 )
278  std::cout << "\nNO RLT REDUCTION CONSTRAINTS FOUND\n";
279  return;
280  }
281 
282  // Display "raw" reduction constraints
283  if( options.DISPLAY >= 2 ){
284  std::cout << "\nRLT REDUCTION CONSTRAINTS:\n";
285  for( auto it = _RLTRed.begin(); it != _RLTRed.end(); ++it ){
286  std::cout << " " << *(*it).first.first << " <";
287  switch( (*it).first.second ){
288  case FFOp::TIMES:
289  std::cout << "MULT"; break;
290  case FFOp::DIV:
291  std::cout << "DIV"; break;
292  default:
293  throw std::runtime_error( "Internal error" );
294  }
295  std::cout << "> " << *(*it).second->pres << " := " << *(*it).second << std::endl;
296  }
297  }
298 
299  // Coefficient matrix and right-hand-side vector holding the "hidden" linear relationships
300  const unsigned nRLTRed = _RLTRed.size();
301  _coefRed.resize( nRLTRed, nRLTRed ); _coefRed.zero(); // Oversize and initialize to zero
302  for( auto itv=_rhsRed.begin(); itv!=_rhsRed.end(); ++itv ) delete *itv;
303  _rhsRed.assign( nRLTRed, 0 );
304  for( auto itv=_varRed.begin(); itv!=_varRed.end(); ++itv ) delete itv->first;
305  _varRed.clear();
306 
307  // Fill-in coefficient matrix and right-hand-side vector
308  auto itRed = _RLTRed.begin();
309  for( unsigned iRow = 0; itRed != _RLTRed.end(); ++itRed, iRow++ )
310  _process_constraint( iRow, vlhs, itRed->first, itRed->second );
311  if( options.DISPLAY >= 2 ){
312  std::cout << "\nRIGHT-HAND-SIDE DEPENDENTS:\n";
313  for( auto itv=_rhsRed.begin(); itv!=_rhsRed.end(); ++itv )
314  std::cout << " " << *(*itv) << std::endl;
315  std::cout << "\nLEFT-HAND-SIDE NEW VARIABLES:\n";
316  for( auto itv=_varRed.begin(); itv!=_varRed.end(); ++itv )
317  std::cout << " " << itv->second << ": " << *itv->first << std::endl;
318  std::cout << "\nCOEFFICIENT MATRIX:\n" << _coefRed;
319  }
320 
321  // Compute a basis of the left null-space using SVD
322  CPPL::dcovector S;
323  CPPL::dgematrix U, VT;
324  if( _coefRed.dgesvd( S, U, VT ) ) return;
325  if( options.DISPLAY >= 2 )
326  std::cout << "\nSVD OF COEFFICIENT MATRIX:\n" << S << U;
327  unsigned nCtrRed = 0;
328  for( ; nCtrRed<nRLTRed; nCtrRed++ )
329  if( S( nRLTRed-nCtrRed-1 ) > options.SVDTOL ) break;
330 
331  // Build "hidden" relationships
332  for( unsigned i=0; i<nCtrRed; i++ ){
333  FFVar ctr = U( 0, nRLTRed-i-1 ) * *_rhsRed[0];
334  for( unsigned j=1; j<nRLTRed; j++ )
335  ctr += U( j, nRLTRed-i-1 ) * *_rhsRed[j];
336  _ctrRed.push_back( *_dag->Vars().find( &ctr ) );
337  }
338 
339  // Display "hidden" relationships
340  if( options.DISPLAY == 1 )
341  std::cout << std::endl << nCtrRed << " REDUCED RLT CONSTRAINTS\n";
342  if( options.DISPLAY >= 2 ){
343  std::cout << std::endl << nCtrRed << " REDUCED RLT CONSTRAINTS FOUND:";
344  for( unsigned i=0; i<nCtrRed; i++ ) std::cout << " " << *_ctrRed[i];
345  std::cout << std::endl;
346  // Generate list of operation for reduction constraints
347  _dag->output( _dag->subgraph( _ctrRed ) );
348  //std::cout << " " << *_ctrRed[i]
349  // << " := " << U( 0, nRLTRed-i-1 ) << " · " << *_rhsRed[0];
350  //for( unsigned j=1; j<nRLTRed; j++ )
351  // std::cout << " + " << U( j, nRLTRed-i-1 ) << " · " << *_rhsRed[j];
352  //std::cout << std::endl;
353  }

356 inline void
357 RLTRED::_process_constraint
358 ( const unsigned iRow, const std::vector<const FFVar*>& vlhs, const t_RLTVar& RLTVar, const FFOp* linOp )
359 {
360  _rhsRed[iRow] = new FFVar(0.);
361 
362  switch( linOp->type ){
363  case FFOp::PLUS:
364  _process_term( iRow, RLTVar.second, *RLTVar.first, *linOp->plop, 1. );
365  _process_term( iRow, RLTVar.second, *RLTVar.first, *linOp->prop, 1. );
366  break;
367 
368  case FFOp::NEG:
369  _process_term( iRow, RLTVar.second, *RLTVar.first, *linOp->plop, -1. );
370  break;
371 
372  case FFOp::MINUS:
373  _process_term( iRow, RLTVar.second, *RLTVar.first, *linOp->plop, 1. );
374  _process_term( iRow, RLTVar.second, *RLTVar.first, *linOp->prop, -1. );
375  break;
376 
377  case FFOp::SCALE:
378  if( linOp->prop->cst() )
379  _process_term( iRow, RLTVar.second, *RLTVar.first, *linOp->plop, linOp->prop->num().val() );
380  else
381  _process_term( iRow, RLTVar.second, *RLTVar.first, *linOp->prop, linOp->plop->num().val() );
382  break;
383 
384  default:
385  throw std::runtime_error( "Internal error" );
386  }
387 
388  if( !_is_rhs( vlhs, *linOp->pres ) )
389  _process_term( iRow, RLTVar.second, *RLTVar.first, *linOp->pres, -1. );
390 }

392 inline void
393 RLTRED::_process_term
394 ( const unsigned iRow, const FFOp::TYPE& RLTOp, const FFVar& RLTVar,
395  const FFVar& linVar, const double coef )
396 {
397  bool foundTerm = false;
398  switch( RLTOp ){
399  // Multiplier variable
400  case FFOp::TIMES:
401  // Multiplied by a constant
402  if( linVar.cst() ){
403  *_rhsRed[iRow] += coef * linVar.num().val() * RLTVar;
404  foundTerm = true;
405  }
406  // In a bilinear term
407  for( auto itt=_bilTerms.begin(); !foundTerm && itt!=_bilTerms.end(); ++itt ){
408  if( ( *(*itt)->plop == linVar && *(*itt)->prop == RLTVar )
409  || ( *(*itt)->prop == linVar && *(*itt)->plop == RLTVar ) ){
410  *_rhsRed[iRow] += coef * *(*itt)->pres;
411  foundTerm = true;
412  }
413  }
414  // In a fractional term
415  for( auto itt=_divTerms.begin(); !foundTerm && itt!=_divTerms.end(); ++itt ){
416  if( ( *(*itt)->pres == linVar && *(*itt)->prop == RLTVar )
417  || ( *(*itt)->prop == linVar && *(*itt)->pres == RLTVar ) ){
418  *_rhsRed[iRow] += coef * *(*itt)->plop;
419  foundTerm = true;
420  }
421  }
422  // In a square term
423  for( auto itt=_sqrTerms.begin(); !foundTerm && itt!=_sqrTerms.end(); ++itt ){
424  if( *(*itt)->plop == linVar && *(*itt)->plop == RLTVar ){
425  *_rhsRed[iRow] += coef * *(*itt)->pres;
426  foundTerm = true;
427  }
428  }
429  // In a square-root term
430  for( auto itt=_sqrtTerms.begin(); !foundTerm && itt!=_sqrtTerms.end(); ++itt ){
431  if( *(*itt)->pres == linVar && *(*itt)->pres == RLTVar ){
432  // Add term to rhs
433  *_rhsRed[iRow] += coef * *(*itt)->plop;
434  foundTerm = true;
435  }
436  }
437  break;
438 
439  // Divider variable
440  case FFOp::DIV:
441  // Divided by itself
442  if( linVar == RLTVar ){
443  *_rhsRed[iRow] += coef;
444  foundTerm = true;
445  }
446  // In a bilinear term
447  for( auto itt=_bilTerms.begin(); !foundTerm && itt!=_bilTerms.end(); ++itt ){
448  if( *(*itt)->pres == linVar && *(*itt)->plop == RLTVar ){
449  *_rhsRed[iRow] += coef * *(*itt)->prop;
450  foundTerm = true;
451  }
452  }
453  // In a fractional term
454  for( auto itt=_divTerms.begin(); !foundTerm && itt!=_divTerms.end(); ++itt ){
455  if( *(*itt)->plop == linVar && *(*itt)->prop == RLTVar ){
456  *_rhsRed[iRow] += coef * *(*itt)->pres;
457  foundTerm = true;
458  }
459  }
460  // In a square term
461  for( auto itt=_sqrTerms.begin(); !foundTerm && itt!=_sqrTerms.end(); ++itt ){
462  if( *(*itt)->pres == linVar && *(*itt)->plop == RLTVar ){
463  *_rhsRed[iRow] += coef * *(*itt)->plop;
464  foundTerm = true;
465  }
466  }
467  // In a square-root term
468  for( auto itt=_sqrtTerms.begin(); !foundTerm && itt!=_sqrtTerms.end(); ++itt ){
469  if( *(*itt)->plop == linVar && *(*itt)->pres == RLTVar ){
470  *_rhsRed[iRow] += coef * *(*itt)->pres;
471  foundTerm = true;
472  }
473  }
474  break;
475 
476  default:
477  throw std::runtime_error( "Internal error" );
478  }
479  if( foundTerm ) return;
480 
481  // Insert or retreive new operation
482  FFOp*newOp = new FFOp( RLTOp, const_cast<FFVar*>(&linVar), const_cast<FFVar*>(&RLTVar) );
483  auto itv = _varRed.insert( std::make_pair( newOp, _varRed.size() ) );
484  if( !itv.second ) delete newOp;
485  const unsigned jCol = itv.first->second;
486  // Assign corresponding entry in coefficent matrix
487  _coefRed( iRow, jCol ) = -coef;
488 }

535 inline void
536 RLTRED::_search_sequential
537 ( const typename RLTRED::l_Ops& Ops )
538 {
539  _RLTRed.clear();
540 
541  // MAIN LOOP: Valid RRLT cut for each candidate multiplier/divider variables
542  for( auto ito = Ops.rbegin(); ito != Ops.rend(); ++ito ){
543  assert( (*ito)->pres );
544  const pt_idVar& idVar = (*ito)->pres->id();
545 
546  // Multiplying/dividing constraints by a constant is pointless
547  if( idVar.first == FFVar::CINT || idVar.first == FFVar::CREAL || (*ito)->pres->cst() ) continue;
548 
549  // Multiplier/divider variables limited to primary variables (no auxiliary)
550  if( options.LEVEL == Options::PRIMSEQ && idVar.first == FFVar::AUX ) continue;
551 
552  // Drop any linear variable participating in a nonlinear terms multiplied by variable <a>idVar</a> and the corresponding edges
553  auto VarMult = std::make_pair( (*ito)->pres, FFOp::TIMES );
554  _RLTVars = _linVars;
555  _RLTEdges = _linEdges;
556  _bigraph_RLT( VarMult, _RLTVars, _RLTEdges );
557  // Identify valid reduction constraints with the candidate multiplier variable
558  _reduction_RLT( VarMult );
559  //break;
560 
561  if( options.NODIV ) continue;
562  // Drop any linear variable participating in a nonlinear terms divided by variable <a>idVar</a> and the corresponding edges
563  auto VarDiv = std::make_pair( (*ito)->pres, FFOp::DIV );
564  _RLTVars = _linVars;
565  _RLTEdges = _linEdges;
566  _bigraph_RLT( VarDiv, _RLTVars, _RLTEdges );
567  // Identify valid reduction constraints with the candidate divider variable
568  _reduction_RLT( VarDiv );
569  }
570 }

595 inline void
596 RLTRED::_bigraph_RLT_mult
597 ( const FFVar*VarMult, t_Vars &linVars, t_Edges &linEdges )
598 const
599 {
600  // Erase constants since their multiplication with <a>idMult</a> does not introduce a new nonlinear term
601  for( auto it = linVars.begin(); it != linVars.end(); ){
602  if( (*it)->id().first != FFVar::CINT && (*it)->id().first != FFVar::CREAL
603  && !(*it)->cst() ) ++it;
604  else{ linEdges.erase( *it ); it = linVars.erase( it ); }
605  }
606 
607  // Erase those variables participating with <a>idMult</a> in bilinear terms
608  for( auto it = _bilTerms.begin(); it != _bilTerms.end(); ++it ){
609  if( *(*it)->plop == *VarMult ){
610  linVars.erase( (*it)->prop ); linEdges.erase( (*it)->prop );
611  }
612  else if( *(*it)->prop == *VarMult ){
613  linVars.erase( (*it)->plop ); linEdges.erase( (*it)->plop );
614  }
615  }
616 
617  // Erase those variables participating with <a>idMult</a> in fractional terms
618  for( auto it = _divTerms.begin(); it != _divTerms.end(); ++it ){

710 ( const t_RLTVar& VarRed )
711 {
712  // Matching initially empty
713  FFOp* NA = 0;
714  _VarAssigned.assign( _RLTVars.size(), std::make_pair( NA, 0 ) );

725  for( unsigned jo=0; jto != _linTerms.end(); ++jto, jo++ )

730 inline bool
731 RLTRED::_augpath_RLT
732 ( const pt_Op& pOp )
733 {
734  _TermVisited[pOp.second] = true;
735 
736  // Try and find immediate assignment
737  auto itv = _RLTVars.begin();
738  for( unsigned iv=0; itv != _RLTVars.end(); ++itv, iv++ ){
739  if( _VarAssigned[iv].first ) continue;
740  auto rangeOp = _RLTEdges.equal_range( *itv );
741  for( auto itvo = rangeOp.first; itvo != rangeOp.second; ++itvo ){
742  if( (*itvo).second != pOp.first ) continue;
743  _VarAssigned[iv] = pOp;
744  return true;
745  }
746  }
747 
748  // Try and find an augmenting path starting from another variable
749  itv = _RLTVars.begin();
750  for( unsigned iv=0; itv != _RLTVars.end(); ++itv, iv++ ){
751  if( _VarVisited[iv] ) continue;
752  auto rangeOp = _RLTEdges.equal_range( *itv );
753  for( auto itvo = rangeOp.first; itvo != rangeOp.second; ++itvo ){
754  if( (*itvo).second != pOp.first ) continue;
755  _VarVisited[iv] = true;
756  if( !_augpath_RLT( _VarAssigned[iv] ) ) continue;
757  _VarAssigned[iv] = pOp;
758  return true;
759  }
760  }
761 
762  // Failed to find an augmenting path
763  return false;
764 }
