34 template <class T>
35 struct lt_expmon
36 {
37  typedef std::pair< unsigned, std::map< unsigned, unsigned > > t_expmon;
38 
39  bool operator()
40  ( const t_expmon&Exp1, const t_expmon&Exp2 ) const
41  {
42  // Order exponents based on their total order first
43  if( Exp1.first < Exp2.first ) return true;
44  if( Exp1.first > Exp2.first ) return false;
45  // Account for the case of an empty list
46  if( Exp1.second.empty() ) return true;
47  if( Exp2.second.empty() ) return false;
48  // Order in lexicographic order next
49  for( auto it1=Exp1.second.begin(), it2=Exp2.second.begin(); it1!=Exp1.second.end(); ++it1, ++it2 ){
50  if( it1->first < it2->first ) return true;
51  if( it1->first > it2->first ) return false;
52  if( it1->second > it2->second ) return true;
53  if( it1->second < it2->second ) return false;
54  }
55  return false;
56  }
57 };

444 template <typename T> inline void
445 SPolyVar<T>::_center()
446 {
447  const double remmid = Op<T>::mid(_bndrem);
448  if( remmid == 0. ) return;
449  if( _coefmon.empty() || _coefmon.begin()->first.first )
450  _coefmon.insert( std::make_pair( std::make_pair( 0, std::map<unsigned,unsigned>() ), remmid ) );
451  else
452  _coefmon.begin()->second += remmid;
453  _bndrem -= remmid;
454  if( _bndpol ) *_bndpol += remmid;
455 }
