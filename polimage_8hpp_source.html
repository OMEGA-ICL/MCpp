569  bool operator()
570  ( const PolVar<T>*Var1, const PolVar<T>*Var2 ) const
571  {
572  // Order variables/constants w.r.t. their types first
573  if( Var1->_id.first < Var2->_id.first ) return true;
574  if( Var1->_id.first > Var2->_id.first ) return false;
575  // If variables, order w.r.t. their index next
576  switch( Var1->_id.first ){
577  case PolVar<T>::VARCONT: case PolVar<T>::VARINT:
578  case PolVar<T>::AUXCONT: case PolVar<T>::AUXINT:
579  if( Var1->_id.second < Var2->_id.second ) return true;
580  if( Var1->_id.second > Var2->_id.second ) return false;
581  break;
582  case PolVar<T>::AUXCST:
583  lt_FFNum ltNum;
584  return ltNum( &Var1->_var.num(), &Var2->_var.num() );
585  break;
586  }
587  return false;
588  }

672 template <typename T>

674 PolVar<T>::add_breakpt
675 ( const double bkpt )
676 {
677  if( !_img ) return;
678  auto itVar = _img->_Vars.find( &_var );
679  if( itVar == _img->_Vars.end() ) return;
680  double atol = _img->options.BREAKPOINT_ATOL, rtol = _img->options.BREAKPOINT_RTOL;
681  if( isequal( Op<T>::l(itVar->second->range()), bkpt, atol, rtol )
682  || isequal( Op<T>::u(itVar->second->range()), bkpt, atol, rtol ) ) return;
683  auto itL = _breakpts.lower_bound( bkpt );
684  if( itL!=_breakpts.end() && isequal( *itL, bkpt, atol, rtol ) ) return;
685  auto itU = _breakpts.upper_bound( bkpt );
686  if( itU!=_breakpts.end() && isequal( *itU, bkpt, atol, rtol ) ) return;
687  itVar->second->_breakpts.insert( bkpt );
688  itVar->second->reset_subdiv();
689  _breakpts.insert( bkpt );
690  reset_subdiv();
691 }

697 {

700  _push_subdiv( XL );

711 inline const std::vector< PolVar<T> >&
712 PolVar<T>::BIN_subdiv
713 ( FFOp*pOp, const bool reset ) const
714 {
715  if( !reset && !_subdiv.second.empty() ) return _subdiv.second;
716  if( reset && !_subdiv.second.empty() ) _subdiv.second.clear();
717  if( !_img ) return _subdiv.second;
718 
719  const unsigned nsubint = _subdiv.first.size()-1;
720  double coef[nsubint];
721  for( unsigned isub=0; isub<nsubint; isub++ ){
722  coef[isub] = _subdiv.first[isub] - _subdiv.first[isub+1];
723  _push_subdiv( PolVar<T>( _img, Op<T>::zeroone(), true ) );
724  }
725  _img->_append_cut( pOp, PolCut<T>::EQ, _subdiv.first[0], nsubint, _subdiv.second.data(), coef, *this, 1. );
726 
727  for( unsigned isub=0; isub<nsubint-1; isub++ ){
728  _push_subdiv( PolVar<T>( _img, Op<T>::zeroone(), false ) );
729  _img->_append_cut( pOp, PolCut<T>::LE, 0., _subdiv.second[nsubint+isub], 1., _subdiv.second[isub], -1. );
730  _img->_append_cut( pOp, PolCut<T>::GE, 0., _subdiv.second[nsubint+isub], 1., _subdiv.second[isub+1], -1. );
731  }
732 
733  return _subdiv.second;
