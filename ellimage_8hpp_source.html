744 template <class T>
745 inline
746 EllImg<T>&
747 EllImg<T>::_set
748 ( const CPPL::dssmatrix& depmap )
749 {
750  _nx = Q().n ; // resets the number of dependent variables to the dimension of the shape matrix provided
751  _current_row = _nx ; // resets _current_row to the number of dependent variables
752  _ip = -1 ; // resets the row index for the product to the
753  _CMpt = 0 ;
754 
755  // No depmap available
756  if( !depmap.n ){
757  // set Ellipsoid to E0(Q0,q0)
758  _Q = Q().to_dssmatrix();
759  _q = c();
760  _depmap_flag = false;
761  _depmap.clear();
762  }
763 
764  //depmap available
765  else{
766  _Q.resize( depmap.n );
767  _q.resize( depmap.n );
768  _q.zero();
769  _depmap_flag = true;
770  _depmap = depmap;
771  // Set the underlying lifted ellipsoid
772  for( long j=0; j<Q().n; ++j ){
773  // Initialise the varindex-th row of _Q with the varindex-th row of the shape matrix of Ell ;
774  for (long i=0; i<=j; ++i ) _Q.put( i, j, Q(i,j) );
775  // Initialise the varindex-th row of _q with the varindex-th row of the centre vector of Ell ;
776  _q(j) = c(j);
777  }
778  }
779 
780  return *this;
781 }

897 template <class T>
898 inline
899 typename EllImg<T>::lin_param
900 EllImg<T>::_linearize_sqr
901 ( const T& domain )
902 {
903  if( options.CHEBUSE ) return _cmodel_linear( sqr , domain );
904 
905  // Compute linear approximation of sqr(m+r*x) with x \in [-1,1] using minimax linearisation
906  double c0, c1, eta;
907  double m = Op<T>::mid( domain );
908  double r = 0.5 * Op<T>::diam( domain );
909  double c_sec=0., x_tan=0., c_tan=0.;
910  c_sec = m*m+r*r;
911  c1 = 2.0*m*r;
912  x_tan = (c1-2.*m*r)/(2.*r*r);
913  c_tan = sqr(m+r*x_tan) - c1*x_tan;
914  eta = 0.5*(c_sec - c_tan);
915  c0 = c_sec - eta;
916  c1 = c1 / r;
917  return std::make_tuple( c0, c1, eta );
918 }
