1051 template <typename T> inline void
1052 CModel<T>::_set_prodmon_exp
1053 ( unsigned int&kvar, const unsigned int*iexp, const unsigned iord,
1054  unsigned int*iprodexp, unsigned int&nprodmon )
1055 {
1056  if( kvar == _nvar || !nprodmon ) return;
1057 
1058  int nprodnew = 0;
1059  for( unsigned imon=0; imon<nprodmon; imon++ ){
1060  // Product with Chebyshev basis function T0
1061  if( !iexp[kvar] || !iprodexp[imon*(_nvar+1)+kvar+1] ){
1062  iprodexp[imon*(_nvar+1)] += iprodexp[imon*(_nvar+1)+kvar+1] + iexp[kvar];
1063  if( iprodexp[imon*(_nvar+1)] <= _nord )
1064  iprodexp[imon*(_nvar+1)+kvar+1] += iexp[kvar];
1065  }
1066  // Product with Chebyshev basis function Ti, i>0
1067  else{
1068  // Only add term if resulting monic index remain less than or equal to _nord
1069  if( iprodexp[imon*(_nvar+1)] + iprodexp[imon*(_nvar+1)+kvar+1] + iexp[kvar] <= _nord ){
1070  for( unsigned ivar=0; ivar<=_nvar; ivar++ )
1071  iprodexp[(nprodmon+nprodnew)*(_nvar+1)+ivar] = iprodexp[imon*(_nvar+1)+ivar];

1073  iprodexp[(nprodmon+nprodnew)*(_nvar+1)+kvar+1] += iexp[kvar];
1074  nprodnew++;
1075  }
1076  if( iexp[kvar] <= iprodexp[imon*(_nvar+1)+kvar+1] ){
1077  iprodexp[imon*(_nvar+1)] += iprodexp[imon*(_nvar+1)+kvar+1] - iexp[kvar];

1079  iprodexp[imon*(_nvar+1)+kvar+1] -= iexp[kvar];
1080  }
1081  else{

1083  if( iprodexp[imon*(_nvar+1)] <= _nord )
1084  iprodexp[imon*(_nvar+1)+kvar+1] = iexp[kvar] - iprodexp[imon*(_nvar+1)+kvar+1];
1085  }
1086  }
1087  }
1088  nprodmon += nprodnew;
1089  return _set_prodmon_exp( ++kvar, iexp, iord, iprodexp, nprodmon );

1264 template <typename T> inline void
1265 CModel<T>::_interpolation
1266 ( double*coefmon, const unsigned nord, const T&X, puniv f )
1267 {
1268  double b( Op<T>::mid(X) ), a( Op<T>::u(X)-b ), x[nord+1], fx[nord+1];
1269  double mulconst( PI/(2.*double(nord+1)) );
1270  for( unsigned i(0); i<=nord; i++ ){
1271  x[i] = std::cos(mulconst*(2.*double(i)+1.));
1272  fx[i] = f( a*x[i]+b );
1273  }
1274 
1275  switch( nord ){
1276  case 0:
1277  coefmon[0] = fx[0];
1278  return;
1279  case 1:
1280  coefmon[0] = 0.5 * ( fx[0] + fx[1] );
1281  coefmon[1] = ( fx[1] - fx[0] ) / ( x[1] - x[0] );
1282  return;
1283  default:
1284  for( unsigned i(0); i<=nord; i++ ){
1285  double mulconst2( std::cos(mulconst*double(i)) ),
1286  mulconst3( 4*std::pow(mulconst2,2)-2 ),
1287  b0( 0 ), b1( 0 );
1288  b0 = fx[nord];
1289  b1 = fx[nord-1] + mulconst3*b0;
1290  for( unsigned j=nord-2; j>1; j-=2 ){
1291  b0 = fx[j] + mulconst3*b1 - b0;
1292  b1 = fx[j-1] + mulconst3*b0 - b1;
1293  }
1294  if( !(nord%2) )
1295  b0 = fx[0] + mulconst3*b1 - b0 - b1;
1296  else{
1297  b0 = fx[1] + mulconst3*b1 - b0;
1298  b0 = fx[0] + mulconst3*b0 - b1 - b0;
1299  }
1300  coefmon[i] = 2./double(nord+1)*mulconst2*b0;
1301  }
1302  coefmon[0] *=0.5;
1303  return;
1304  }
